* zsh

  #+BEGIN_VERSE
  1. export HISTSIZE=0: 不记录命令历史
  2. unsetopt share_history: session间不共享历史记录
  #+END_VERSE

* no moduled named _ctypes
  #+BEGIN_VERSE
  apt-get install libffi-dev
  #+END_VERSE
* no such module gnutls while compile emacs
  #+BEGIN_VERSE
  apt-get install libffi-dev
  #+END_VERSE
* makeinfo command not found
  #+BEGIN_VERSE
  sudo apt-get install texinfo
  #+END_VERSE
* chacha20-ietf-poly1305 not supported
  #+BEGIN_VERSE
  apt install libsodium-dev
  pip3 install https://github.com/shadowsocks/shadowsocks/archive/master.zip -U
  #+END_VERSE
* mail command not found
  #+BEGIN_VERSE
  yum install -y <mailx
  apt install -y mailutils
  #+END_VERSE
* no pip
  #+BEGIN_VERSE
  curl https://bootstrap.pypa.io/get-pip.py | /opt/python2.7/bin/python
  #+END_VERSE
